-- 1. Seleziona tutti gli utenti e calcolane l'età (25)
SELECT id, username, TIMESTAMPDIFF(YEAR, birthdate, CURDATE()) AS age
FROM users;

--  2. Seleziona tutti i post senza Like (13)
SELECT *
FROM posts
LEFT JOIN likes ON posts.id = likes.post_id
WHERE likes.post_id IS NULL;

-- 3. Conta il numero di like per ogni post (165)
SELECT posts.id, COUNT(likes.user_id) AS likes
FROM posts
LEFT JOIN likes ON likes.post_id = posts.id
GROUP BY posts.id;

-- 4. Ordina gli utenti per il numero di media caricati (25)
SELECT users.id, username, COUNT(posts.id) AS numeri_di_post
FROM users
JOIN posts ON users.id = posts.user_id
GROUP BY users.id
ORDER BY numeri_di_post;

-- 5. Ordina gli utenti per totale di likes ricevuti nei loro posts (25)
SELECT users.id, COUNT(post_id) AS numeri_di_LIKE
FROM users
JOIN likes ON users.id = likes.user_id
GROUP BY user_id
ORDER BY numeri_di_LIKE ASC;

-- 6. Seleziona tutti i post degli utenti tra i 20 e i 30 anni (49) 🙅‍♂️
SELECT posts.*
FROM posts
JOIN users ON users.id = posts.user_id
WHERE TIMESTAMPDIFF(YEAR, users.birthdate, CURDATE()) BETWEEN 20 AND 30;

-- 7. Seleziona il numero di post e di media per ogni utente
SELECT 
  users.id,
  username,
  COUNT(DISTINCT posts.id) AS numero_post,
  COUNT(DISTINCT medias.id) AS numero_media
FROM users
LEFT JOIN posts ON posts.user_id = users.id
LEFT JOIN medias ON medias.user_id = users.id
GROUP BY users.id, username
ORDER BY users.id;
