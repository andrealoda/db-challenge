-- 1. Seleziona tutti gli utenti e calcolane l'età (25)
SELECT id, username, TIMESTAMPDIFF(YEAR, birthdate, CURDATE()) AS age
FROM users;

--  2. Seleziona tutti i post senza Like (13)
SELECT *
FROM posts
LEFT JOIN likes ON posts.id = likes.post_id
WHERE likes.post_id IS NULL;

-- 3. Conta il numero di like per ogni post (165)
SELECT posts.id, COUNT(likes.user_id) AS likes
FROM posts
LEFT JOIN likes ON likes.post_id = posts.id
GROUP BY posts.id;

-- 4. Ordina gli utenti per il numero di media caricati (25)
SELECT users.id, username, COUNT(posts.id) AS numeri_di_post
FROM users
JOIN posts ON users.id = posts.user_id
GROUP BY users.id
ORDER BY numeri_di_post;
